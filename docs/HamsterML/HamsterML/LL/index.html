type ll_expr =
  1. | LLConst of Ast.value
  2. | LLVar of string
  3. | LLOperation of Ast.op
  4. | LLTuple of ll_expr * ll_expr * ll_expr list
  5. | LLList of ll_expr list
  6. | LLListConcat of ll_expr * ll_expr
  7. | LLConstraint of ll_expr * Ast.dataType
  8. | LLApplication of ll_expr * ll_expr
  9. | LLLet of Ast.funType * ll_bind list * ll_expr option
  10. | LLIf of ll_expr * ll_expr * ll_expr option
  11. | LLMatch of ll_expr * ll_case list
and ll_bind = Ast.pattern * Ast.args * ll_expr
and ll_case = Ast.pattern * ll_expr
and ll_prog = ll_expr list
