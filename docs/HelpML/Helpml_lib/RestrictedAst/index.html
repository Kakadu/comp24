Module Helpml_lib.RestrictedAst

type immexpr =
  1. | ImmNum of int
  2. | ImmId of Ast.id
  3. | ImmBool of bool
  4. | ImmUnit
type pexpr =
  1. | PImmExpr of immexpr
  2. | PImmWild
type cexpr =
  1. | CBinOp of Ast.bin_op * immexpr * immexpr
  2. | CApp of immexpr * immexpr
  3. | CImmExpr of immexpr
  4. | CIf of immexpr * aexpr * aexpr
and aexpr =
  1. | ALetIn of Ast.id * cexpr * aexpr
  2. | ACExpr of cexpr
type bexpr =
  1. | ALet of bool * Ast.id * pexpr list * aexpr
type prexpr = bexpr list
