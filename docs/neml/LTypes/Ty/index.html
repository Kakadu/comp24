type t =
  1. | Var of Var.t
    (*

    A type variable such as 'a

    *)
  2. | Arr of t * t
    (*

    T1 -> T2

    *)
  3. | Tuple of t LMisc.List2.t
    (*

    T1 * ... * Tn

    *)
  4. | Con of LMisc.Id.t * t Base.list
    (*

    Con(tconstr, l) represents:

    • tconstr when l=[]
    • T tconstr when l=[T]
    • (T1, ..., Tn) tconstr when l=[T1, ..., Tn]
    *)
