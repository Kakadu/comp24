Module LAst.Pat

type t =
  1. | Any
    (*

    The pattern _

    *)
  2. | Var of LMisc.Id.t
    (*

    A variable pattern such as x

    *)
  3. | Const of Const.t
    (*

    Patterns such as 1, 'a', "hello", 1.5

    *)
  4. | Tuple of t LMisc.List2.t
    (*

    (P1, ..., Pn)

    *)
  5. | Or of t * t
    (*

    P1 | P2

    *)
  6. | Construct of LMisc.Id.t * t Base.option
    (*

    Construct(C, arg) represents:

    • C when arg is None
    • C P when arg is Some P
    *)
  7. | Constraint of t * LTypes.Ty.t
    (*

    (P : T)

    *)
val pp : Ppx_deriving_runtime.Format.formatter -> t -> Ppx_deriving_runtime.unit
val show : t -> Ppx_deriving_runtime.string
