type value_binding = {
  1. pat : Pat.t;
  2. expr : t;
}
and t =
  1. | Id of LMisc.Id.t
    (*

    Identifiers such as x, fact

    *)
  2. | Const of Const.t
    (*

    Expression constant such as 1, 'a', "hello", 1.5

    *)
  3. | Let of rec_flag * value_binding LMisc.List1.t * t
    (*

    Let(flag, [(P1,E1) ; ... ; (Pn,En)], E) represents:

    • let P1 = E1 and ... and Pn = EN in E when flag is Nonrec
    • let rec P1 = E1 and ... and Pn = EN in E when flag is Rec
    *)
  4. | Fun of Pat.t LMisc.List1.t * t
    (*

    fun P1 ... Pn -> E

    *)
  5. | Function of value_binding LMisc.List1.t
    (*

    function C1 | ... | Cn

    *)
  6. | Apply of t * t
    (*

    E1 E2

    *)
  7. | Match of t * value_binding LMisc.List1.t
    (*

    match E with P1 -> E1 | ... | Pn -> En

    *)
  8. | Tuple of t LMisc.List2.t
    (*

    (E1, ..., En)

    *)
  9. | Construct of LMisc.Id.t * t Base.option
    (*

    Construct(C, exp) represents:

    • C when exp is None
    • C E when exp is Some E
    • C (E1, ..., En) when exp is Some (Tuple[E1,...,En])
    *)
  10. | If of t * t * t Base.option
    (*

    if E1 then E2 else E3

    *)
  11. | Seq of t LMisc.List2.t
    (*

    E1; E2

    *)
  12. | Constraint of t * LTypes.Ty.t
    (*

    (E : T)

    *)
val pp_value_binding : Ppx_deriving_runtime.Format.formatter -> value_binding -> Ppx_deriving_runtime.unit
val show_value_binding : value_binding -> Ppx_deriving_runtime.string
val pp : Ppx_deriving_runtime.Format.formatter -> t -> Ppx_deriving_runtime.unit
val show : t -> Ppx_deriving_runtime.string
