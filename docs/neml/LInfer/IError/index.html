Module LInfer.IError

type t =
  1. | UnboundVariable of LMisc.Id.t
    (*

    Failed to find a variable in the environment

    *)
  2. | UnificationFail of LTypes.Ty.t * LTypes.Ty.t
    (*

    Failed to unify two types

    *)
  3. | UnificationMismatch of LTypes.Ty.t Base.list * LTypes.Ty.t Base.list
    (*

    Lists of types to unify have different lengths

    *)
  4. | UnboundType of LMisc.Id.t
    (*

    Type is unbound in type declaration. E.g. `type bar = Bar of foo` (foo is unbound)

    *)
  5. | UnboundTypeVariable of LTypes.Var.t
    (*

    Type variable is unbound in type declaration. E.g. `type bar = Bar of 'a` ('a is unbound)

    *)
  6. | TypeArityMismatch of LMisc.Id.t
    (*

    Type expects more/less arguments than applied. E.g. `type foo = Foo` (foo expects 0 arguments) `type bar = Bar of int foo`

    *)
  7. | OccursIn of LTypes.Var.t * LTypes.Ty.t
    (*

    Type variable occurs in a type

    *)
  8. | PatVarBoundSeveralTimes of LMisc.Id.t
    (*

    Pattern(s) bound the same variable several times. E.g. `let x, x = ...`

    *)
  9. | NotVarLHSRec of LAst.Pat.t
    (*

    The left hand side of the recursive binding is not a var. E.g. `let rec _ = ...`

    *)
  10. | NotAllowedRHSRec of LAst.Expr.t
    (*

    The expression is not allowed on the right-hand side of `let rec'. E.g. `let rec x = x + 1`

    *)
  11. | NotImplemented of Base.string
    (*

    Too bad something's not done

    *)
val pp : Ppx_deriving_runtime.Format.formatter -> t -> Ppx_deriving_runtime.unit
val show : t -> Ppx_deriving_runtime.string
