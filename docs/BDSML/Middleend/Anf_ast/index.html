type aexpr =
  1. | AExp_ident of string
  2. | AExp_constant of Parser.Ast.constant
  3. | AExp_tuple of aexpr list
  4. | AExp_construct of string * aexpr option
type cexpr =
  1. | CExp_if of aexpr * lexpr * lexpr
  2. | CExp_apply of string * aexpr list
  3. | CExp_atom of aexpr
and lexpr =
  1. | LLet_in of string * cexpr * lexpr
  2. | LComplex of cexpr
type func = string * string list * lexpr
type absexpr =
  1. | AbsStr_func of func
  2. | AbsStr_value of string * lexpr
  3. | AbsStr_value_rec of func list
type anf = absexpr list
