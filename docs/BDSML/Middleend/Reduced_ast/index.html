type rexpr =
  1. | RExp_ident of string
    (*

    like Exp_ident

    *)
  2. | RExp_constant of Parser.Ast.constant
    (*

    like Exp_constant

    *)
  3. | RExp_let of string * rexpr * rexpr
    (*

    e.g. let a = b in c

    *)
  4. | RExp_let_rec of (string * rexpr) list * rexpr
    (*

    e.g. let rec a = b and c = a in d

    *)
  5. | RExp_fun of string list * rexpr
    (*

    like Exp_fun but only non pattern args

    *)
  6. | RExp_apply of rexpr * rexpr
    (*

    like Exp_apply

    *)
  7. | RExp_tuple of rexpr list
    (*

    like Exp_tuple

    *)
  8. | RExp_construct of string * rexpr option
    (*

    like Exp_construct

    *)
  9. | RExp_if of rexpr * rexpr * rexpr option
    (*

    like Exp_if

    *)
type rstuct_item =
  1. | RStr_eval of rexpr
  2. | RStr_value of string * rexpr
  3. | RStr_value_rec of (string * rexpr) list
type rstruct = rstuct_item list
type error =
  1. | Invalid_pattern of string
  2. | Invalid_ast of string
  3. | Invalid_previous_result of string * string
