type constant =
  1. | Const_int of int
    (*

    Integer literal, e.g. 69

    *)
  2. | Const_char of char
    (*

    Character literal, e.g. 'm'

    *)
  3. | Const_string of string
    (*

    String literal, e.g. "something"

    *)
  4. | Const_bool of bool
    (*

    Bool literal, true or false

    *)
  5. | Const_unit
    (*

    Unit literal, ()

    *)
val pp_constant : Ppx_deriving_runtime.Format.formatter -> constant -> Ppx_deriving_runtime.unit
val show_constant : constant -> Ppx_deriving_runtime.string
type typexpr =
  1. | Type_constructor_param of typexpr * string
    (*

    e.g. int list

    *)
  2. | Type_tuple of typexpr list
    (*

    e.g. int * int

    *)
  3. | Type_single of string
    (*

    e.g. int

    *)
  4. | Type_fun of typexpr list
    (*

    e.g. int -> int

    *)
val pp_typexpr : Ppx_deriving_runtime.Format.formatter -> typexpr -> Ppx_deriving_runtime.unit
val show_typexpr : typexpr -> Ppx_deriving_runtime.string
type rec_flag =
  1. | Nonrecursive
  2. | Recursive
val pp_rec_flag : Ppx_deriving_runtime.Format.formatter -> rec_flag -> Ppx_deriving_runtime.unit
val show_rec_flag : rec_flag -> Ppx_deriving_runtime.string
type pattern =
  1. | Pat_any
    (*

    Pattern any _

    *)
  2. | Pat_var of string
    (*

    Var pattern, e.g. x

    *)
  3. | Pat_type of pattern * typexpr
    (*

    Pattern with type, e.g. x:int

    *)
  4. | Pat_constant of constant
    (*

    Constant patterns, e.g. 69, 'm', "something"

    *)
  5. | Pat_tuple of pattern list
    (*

    Pattern for many elements, e.g. P1, ..., Pn (n >= 2)

    *)
  6. | Pat_or of pattern * pattern
    (*

    Pattern for one of elements, e.g. P1 | P2

    *)
  7. | Pat_construct of string * pattern option
    (*

    Pat_construct(C, args) represents:

    • C when args is None
    • C P when args is Some P
    *)
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type let_binding =
  1. | Pat_binding of pattern * expression
    (*

    e.g. let (a, b) = (1, 2)

    *)
  2. | Val_binding of string * pattern list * expression
    (*

    e.g. let f a b = a + b

    *)
and case = {
  1. left : pattern;
  2. right : expression;
}
and expression =
  1. | Exp_ident of string
    (*

    Identifiers, e.g. some_var

    *)
  2. | Exp_constant of constant
    (*

    Expression constant, e.g. 69, 'm', "something"

    *)
  3. | Exp_type of expression * typexpr
    (*

    Expression with type, e.g. 69:int

    *)
  4. | Exp_let of rec_flag * let_binding list * expression
    (*

    Exp_let(flag, [(P1,E1) ; ... ; (Pn,En)], E) represents:

    • let P1 = E1 and ... and Pn = EN in E when flag is Nonrecursive
    • let rec P1 = E1 and ... and Pn = EN in E when flag is Recursive. Invariant: n >= 1
    *)
  5. | Exp_fun of pattern list * expression
    (*

    Exp_fun ([P1; ...; Pn], E) represents fun P1 ... Pn -> E. Invariant: n >= 1

    *)
  6. | Exp_function of case list
    (*

    Exp_function([C1; ...; Cn]) represents function C1 | ... | Cn. Invariant: n >= 1

    *)
  7. | Exp_apply of expression * expression
    (*

    Exp_apply(E0, E1) represents E0 E1

    *)
  8. | Exp_match of expression * case list
    (*

    match E0 with P1 -> E1 | ... | Pn -> En. Invariant: n >= 1

    *)
  9. | Exp_tuple of expression list
    (*

    Expressions (E1, ..., En). Invariant: n >= 2

    *)
  10. | Exp_construct of string * expression option
    (*

    Exp_construct(C, exp) represents:

    • C when exp is None,
    • C E when exp is Some E,
    • C (E1, ..., En) when exp is Some (Exp_tuple[E1;...;En])
    *)
  11. | Exp_if of expression * expression * expression option
    (*

    if E1 then E2 else E3

    *)
val pp_let_binding : Ppx_deriving_runtime.Format.formatter -> let_binding -> Ppx_deriving_runtime.unit
val show_let_binding : let_binding -> Ppx_deriving_runtime.string
val pp_case : Ppx_deriving_runtime.Format.formatter -> case -> Ppx_deriving_runtime.unit
val show_case : case -> Ppx_deriving_runtime.string
val pp_expression : Ppx_deriving_runtime.Format.formatter -> expression -> Ppx_deriving_runtime.unit
val show_expression : expression -> Ppx_deriving_runtime.string
type structure_item =
  1. | Str_eval of expression
    (*

    Expression

    *)
  2. | Str_value of rec_flag * let_binding list
    (*

    Str_value(flag, [(P1, E1) ; ... ; (Pn, En)]) represents:

    • let P1 = E1 and ... and Pn = EN when flag is Nonrecursive
    • let rec P1 = E1 and ... and Pn = EN when flag is Recursive. Invariant: n >= 1
    *)
val pp_structure_item : Ppx_deriving_runtime.Format.formatter -> structure_item -> Ppx_deriving_runtime.unit
val show_structure_item : structure_item -> Ppx_deriving_runtime.string
type structure = structure_item list
val pp_structure : Ppx_deriving_runtime.Format.formatter -> structure -> Ppx_deriving_runtime.unit
val show_structure : structure -> Ppx_deriving_runtime.string
