type const =
  1. | C_int of int
    (*

    1

    *)
  2. | C_bool of bool
    (*

    true

    *)
  3. | C_empty_list
    (*

    *)
  4. | C_unit
    (*

    ()

    *)
val pp_const : Ppx_deriving_runtime.Format.formatter -> const -> Ppx_deriving_runtime.unit
val show_const : const -> Ppx_deriving_runtime.string
type raw_typ =
  1. | RT_var of string
    (*

    type var

    *)
  2. | RT_prim of string
    (*

    ground type

    *)
  3. | RT_arr of raw_typ * raw_typ
    (*

    function type

    *)
  4. | RT_tuple of raw_typ * raw_typ list
    (*

    tuple type

    *)
  5. | RT_list of raw_typ
    (*

    list type

    *)
val pp_raw_typ : Ppx_deriving_runtime.Format.formatter -> raw_typ -> Ppx_deriving_runtime.unit
val show_raw_typ : raw_typ -> Ppx_deriving_runtime.string
type pattern =
  1. | P_typed of pattern * raw_typ
    (*

    (a: int)

    *)
  2. | P_any
    (*

    _

    *)
  3. | P_val of string
    (*

    abc

    *)
  4. | P_const of const
    (*

    1

    *)
  5. | P_tuple of pattern * pattern list
    (*

    (1, 2)

    *)
  6. | P_cons_list of pattern * pattern
    (*

    1:: or 1;2

    *)
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expr =
  1. | E_typed of expr * raw_typ
    (*

    val1 : int

    *)
  2. | E_const of const
    (*

    1

    *)
  3. | E_ident of string
    (*

    val1 or ( + )

    *)
  4. | E_ite of expr * expr * expr
    (*

    if a then b else c

    *)
  5. | E_fun of pattern * pattern list * expr
    (*

    fun x -> x + 1

    *)
  6. | E_app of expr * expr
    (*

    f x

    *)
  7. | E_let of decl * expr
    (*

    let a = 1 in b

    *)
  8. | E_match of expr * (pattern * expr) list
    (*

    match x with 1 -> 0 | _ -> 1

    *)
  9. | E_cons_list of expr * expr
    (*

    1:: or 1;2

    *)
  10. | E_tuple of expr * expr list
    (*

    (1, a)

    *)
and decl_body = pattern * raw_typ option * expr
and decl =
  1. | Non_rec of decl_body
    (*

    let x = 1

    *)
  2. | Rec of decl_body list
    (*

    let rec a = ... and b = ...

    *)
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string
val pp_decl_body : Ppx_deriving_runtime.Format.formatter -> decl_body -> Ppx_deriving_runtime.unit
val show_decl_body : decl_body -> Ppx_deriving_runtime.string
val pp_decl : Ppx_deriving_runtime.Format.formatter -> decl -> Ppx_deriving_runtime.unit
val show_decl : decl -> Ppx_deriving_runtime.string
type toplevel =
  1. | Let_decl of decl
    (*

    let a = 1

    *)
  2. | Expr of expr
    (*

    a + b

    *)
val pp_toplevel : Ppx_deriving_runtime.Format.formatter -> toplevel -> Ppx_deriving_runtime.unit
val show_toplevel : toplevel -> Ppx_deriving_runtime.string
type program = toplevel list
val pp_program : Ppx_deriving_runtime.Format.formatter -> program -> Ppx_deriving_runtime.unit
val show_program : program -> Ppx_deriving_runtime.string
