type rec_flag =
  1. | Rec
    (*

    rec

    *)
  2. | NoRec
    (*

    norec

    *)
type pe_const =
  1. | Pe_Cint of int
    (*

    1

    *)
  2. | Pe_CBool of bool
    (*

    true

    *)
type pe_expr =
  1. | Pe_EUnit
    (*

    ()

    *)
  2. | Pe_ENill
    (*

    *)
  3. | Pe_EIdentifier of string
    (*

    x

    *)
  4. | Pe_EConst of pe_const
    (*

    1 || true

    *)
  5. | Pe_EIf of pe_expr * pe_expr * pe_expr
    (*

    if E1 then E2 else E3

    *)
  6. | Pe_EFun of string list * pe_expr
    (*

    fun x y -> E

    *)
  7. | Pe_EApp of pe_expr * pe_expr
    (*

    E1 E2

    *)
  8. | Pe_ELet of rec_flag * string * pe_expr * pe_expr
    (*

    let (rec) f = E1 in E2

    *)
  9. | Pe_ECons of pe_expr * pe_expr
    (*

    E1 :: E2

    *)
  10. | Pe_ETuple of pe_expr list
    (*

    (E1, E2, E3)

    *)
type pe_declaration =
  1. | Pe_Nonrec of (string * pe_expr) list
    (*

    (let f1 = E1 (and f2 = E2)

    *)
  2. | Pe_Rec of (string * pe_expr) list
    (*

    (let rec f1 = E1 (and f2 = E2)

    *)
type pe_program = pe_declaration list
val const_to_str : pe_const -> string
val expr_to_str : pe_expr -> string
val decl_to_str : pe_declaration -> string
val pp_pe_program : Stdlib.Format.formatter -> pe_program -> unit
