type type_annotation =
  1. | AUnit
  2. | AInt
  3. | ABool
  4. | AList of type_annotation
  5. | ATuple of type_annotation list
  6. | AFunction of type_annotation * type_annotation
val pp_type_annotation : Ppx_deriving_runtime.Format.formatter -> type_annotation -> Ppx_deriving_runtime.unit
val show_type_annotation : type_annotation -> Ppx_deriving_runtime.string
type pattern =
  1. | PNill
    (*

    *)
  2. | PAny
    (*

    _

    *)
  3. | PUnit
    (*

    ()

    *)
  4. | PConst of const
    (*

    1 || true

    *)
  5. | PIdentifier of id
    (*

    x

    *)
  6. | PTuple of pattern list
    (*

    (x, y, z)

    *)
  7. | PCons of pattern * pattern
    (*

    hd :: tl

    *)
  8. | PConstraint of pattern * type_annotation
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expression =
  1. | EUnit
  2. | ENill
  3. | EConstraint of expression * type_annotation
  4. | EConst of const
  5. | EIdentifier of id
  6. | EApplication of expression * expression
    (*

    E1 E2

    *)
  7. | EFun of pattern * expression
    (*

    fun P -> E

    *)
  8. | ELetIn of rec_flag * pattern * expression * expression
    (*

    let f x = E1

    *)
  9. | ETuple of expression list
  10. | EIf of expression * expression * expression
    (*

    if e1 then e2 else e3

    *)
  11. | ECons of expression * expression
    (*

    a; b; c, a :: b; c

    *)
  12. | EMatch of expression * (pattern * expression) list
    (*

    match e with p1 -> e1 |...| pn -> en

    *)
val pp_expression : Ppx_deriving_runtime.Format.formatter -> expression -> Ppx_deriving_runtime.unit
val show_expression : expression -> Ppx_deriving_runtime.string
