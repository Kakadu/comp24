type immexpr =
  1. | ImmInt of int
  2. | ImmIdentifier of string
  3. | ImmBool of bool
  4. | ImmUnit
  5. | ImmNill
  6. | ImmTuple of immexpr list
type cexpr =
  1. | CEApply of string * immexpr list
  2. | CEIf of immexpr * aexpr * aexpr
  3. | CECons of immexpr * immexpr
  4. | CImmExpr of immexpr
and aexpr =
  1. | ALetIn of string * cexpr * aexpr
  2. | ACExpr of cexpr
type anf_binding =
  1. | ALet of string * string list * aexpr
type anf_decl =
  1. | ADNoRec of anf_binding list
  2. | ADREC of anf_binding list
type anf_prog = anf_decl list
val imm_id : string -> immexpr
val cimmexpr : immexpr -> cexpr
val atom_to_str : immexpr -> string
val cexp_to_str : cexpr -> string
val exp_to_str : aexpr -> string
val fun_to_str : anf_binding -> string
val declaration_to_str : anf_decl -> string
val pp_anf_program : Stdlib.Format.formatter -> anf_decl Base.List.t -> unit
