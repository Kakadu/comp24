type pe_const =
  1. | PECint of int
  2. | PECBool of bool
  3. | PECNil
  4. | PECUnit
  5. | PECString of string
type pe_expr =
  1. | PEEConst of pe_const
  2. | PEEVar of string
  3. | PEEIf of pe_expr * pe_expr * pe_expr
  4. | PEEFun of string list * pe_expr
  5. | PEEApp of pe_expr * pe_expr
  6. | PEELet of pe_str_item * pe_expr
  7. | PEECons of pe_expr * pe_expr
  8. | PEETuple of pe_expr list
and pe_str_item =
  1. | PENonrec of string * pe_expr
  2. | PERec of (string * pe_expr) list
type pe_structure = pe_str_item list
val const_to_str : pe_const -> string
val expr_to_str : pe_expr -> string
val decl_to_str : pe_str_item -> string
val pp_pe_expr : Stdlib.Format.formatter -> pe_expr -> unit
val pp_pe_structure : Stdlib.Format.formatter -> pe_str_item list -> unit
type value_to_get =
  1. | Tuple of int
  2. | Cons_head
  3. | Cons_tail
  4. | Other

val const_to_peconst : Ast.const -> pe_expr
