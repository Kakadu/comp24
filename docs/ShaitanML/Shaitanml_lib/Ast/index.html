type id = string
val pp_id : Stdlib.Format.formatter -> id -> unit
val show_id : id -> string
type rec_flag =
  1. | Rec
  2. | Nonrec
val pp_rec_flag : Stdlib.Format.formatter -> rec_flag -> unit
val show_rec_flag : rec_flag -> string
type const =
  1. | CInt of int
  2. | CBool of bool
  3. | CString of id
  4. | CUnit
  5. | CNil
val pp_const : Stdlib.Format.formatter -> const -> unit
val show_const : const -> string
type type_annot =
  1. | AInt
  2. | ABool
  3. | AString
  4. | AUnit
  5. | AList of type_annot
  6. | AFun of type_annot * type_annot
  7. | ATuple of type_annot list
  8. | AVar of id
val pp_type_annot : Stdlib.Format.formatter -> type_annot -> unit
val show_type_annot : type_annot -> string
type pattern =
  1. | PAny
  2. | PConst of const
  3. | PVar of id
  4. | PTuple of pattern list
  5. | PCons of pattern * pattern
  6. | PConstraint of pattern * type_annot
val pp_pattern : Stdlib.Format.formatter -> pattern -> unit
val show_pattern : pattern -> string
type expr =
  1. | EConst of const
  2. | EVar of id
  3. | EIf of expr * expr * expr
  4. | EMatch of expr * case list
  5. | ELet of rec_flag * binding * expr
  6. | EFun of pattern * expr
  7. | ETuple of expr list
  8. | ECons of expr * expr
  9. | EApply of expr * expr
  10. | EConstraint of expr * type_annot
and case = pattern * expr
and binding = pattern * expr
type str_item =
  1. | SEval of expr
  2. | SValue of rec_flag * binding list
type structure = str_item list
val pp_expr : Stdlib.Format.formatter -> expr -> unit
val show_expr : expr -> string
val pp_case : Stdlib.Format.formatter -> case -> unit
val show_case : case -> string
val pp_binding : Stdlib.Format.formatter -> binding -> unit
val show_binding : binding -> string
val pp_str_item : Stdlib.Format.formatter -> str_item -> unit
val show_str_item : str_item -> string
val constr_apply : expr -> expr list -> expr
val constr_fun : pattern list -> expr -> expr
val pp_structure : Stdlib.Format.formatter -> structure -> unit
val show_structure : structure -> string
val print_structure : structure -> string
