type id = string

Identifiers for variables and functions names

val pp_id : Ppx_deriving_runtime.Format.formatter -> id -> Ppx_deriving_runtime.unit
val show_id : id -> Ppx_deriving_runtime.string
type const =
  1. | CInt of int
    (*

    Integers 1, 2, ...

    *)
  2. | CBool of bool
    (*

    Boolean true or false

    *)
  3. | CUnit
    (*

    Unit ()

    *)
val pp_const : Ppx_deriving_runtime.Format.formatter -> const -> Ppx_deriving_runtime.unit
val show_const : const -> Ppx_deriving_runtime.string
type pattern =
  1. | PWild
    (*

    | _ -> ...

    *)
  2. | PEmpty
    (*

    | -> ...

    *)
  3. | PConst of const
    (*

    | const -> ...

    *)
  4. | PVar of id
    (*

    | varname -> ...

    *)
  5. | PCons of pattern * pattern * pattern list
    (*

    | p1 :: p2 -> ...

    *)
  6. | POr of pattern * pattern * pattern list
    (*

    | p1 | p2 | p3 -> ...

    *)
  7. | PTuple of pattern * pattern * pattern list
    (*

    (p1, p2, ...) -> ...

    *)
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type is_rec =
  1. | Rec
    (*

    Recursive functions tag, can call themselves in their body

    *)
  2. | NonRec
    (*

    Non-recursive functions tag

    *)
val pp_is_rec : Ppx_deriving_runtime.Format.formatter -> is_rec -> Ppx_deriving_runtime.unit
val show_is_rec : is_rec -> Ppx_deriving_runtime.string
type type_id =
  1. | TInt
    (*

    Type int

    *)
  2. | TBool
    (*

    Type bool

    *)
  3. | TUnit
    (*

    Type unit ()

    *)
  4. | TFun of type_id * type_id
    (*

    Function type t1 -> t2

    *)
  5. | TList of type_id
    (*

    List type t list

    *)
  6. | TTuple of type_id * type_id * type_id list
    (*

    Tuple type (t1 * t2 * ... * tn)

    *)
val pp_type_id : Ppx_deriving_runtime.Format.formatter -> type_id -> Ppx_deriving_runtime.unit
val show_type_id : type_id -> Ppx_deriving_runtime.string
type typed_arg = id * type_id option
val pp_typed_arg : Ppx_deriving_runtime.Format.formatter -> typed_arg -> Ppx_deriving_runtime.unit
val show_typed_arg : typed_arg -> Ppx_deriving_runtime.string
type expr =
  1. | EConst of const
    (*

    Consts

    *)
  2. | EVar of id
    (*

    Variables with their names

    *)
  3. | ETuple of expr * expr * expr list
    (*

    Tuples of 2 or more elements, separated by ','

    *)
  4. | EList of expr list
    (*

    Lists 1; 2; 3, ...

    *)
  5. | EBranch of expr * expr * expr
    (*

    if cond then a else b

    *)
  6. | EMatch of expr * (pattern * expr) list
    (*

    match x with | p1 -> e1 | ...

    *)
  7. | ELetIn of is_rec * id * expr * expr
    (*

    let rec f: t1 -> t2

    *)
  8. | EFun of typed_arg * expr
    (*

    Anonymous function with typed arguments

    *)
  9. | EApp of expr * expr
    (*

    Application f x y z

    *)
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string
type decl =
  1. | DLet of is_rec * id * expr
  2. | DMutualLet of is_rec * (id * expr) list
val pp_decl : Ppx_deriving_runtime.Format.formatter -> decl -> Ppx_deriving_runtime.unit
val show_decl : decl -> Ppx_deriving_runtime.string
type program = decl list
val pp_program : Ppx_deriving_runtime.Format.formatter -> program -> Ppx_deriving_runtime.unit
val show_program : program -> Ppx_deriving_runtime.string
