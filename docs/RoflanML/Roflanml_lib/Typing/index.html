type base_type =
  1. | BInt
    (*

    Basic integer type

    *)
  2. | BBool
    (*

    Basic bool type

    *)
  3. | BUnit
    (*

    Unit type

    *)
type ty =
  1. | TBase of base_type
    (*

    Type of integer

    *)
  2. | TVar of int
    (*

    Polymorphic type e.g. 'a

    *)
  3. | TArrow of ty * ty
    (*

    Type of function ty1 -> ty2

    *)
  4. | TTuple of ty * ty * ty list
    (*

    Type of tuple

    *)
  5. | TList of ty
    (*

    Type of list

    *)
val map_type : Ast.type_id -> ty
val pp_ty : Stdlib.Format.formatter -> ty -> unit
type error =
  1. | OccursCheckFailed of int * ty
    (*

    OCaml's Occurs check

    *)
  2. | UndeclaredVariable of Ast.id
    (*

    Attempt to use non-initialized variable

    *)
  3. | UnificationFailed of ty * ty
    (*

    Failed to unify left and right types

    *)
  4. | OrPatternBoundsDiff of Ast.id
    (*

    Variable id doesn't occure in some of Or patterns

    *)
  5. | OrPatternTypeDiff of Ast.id * ty * ty
    (*

    Types of some bounds in Or pattern differ

    *)
  6. | NotImplemented
    (*

    Still not implemented features

    *)
  7. | NotReachable
    (*

    For non reachable code. If it is reached, there are serious problems in your typechecker

    *)
val pp_error : Stdlib.Format.formatter -> error -> unit
