type pmf_case = Ast.AbstractSyntaxTree.identifier * pmf_expression

Binding identifier -> expression

and pmf_expression =
  1. | PMFConstant of Ast.AbstractSyntaxTree.constant
    (*

    Any constant: '1', 'true', etc

    *)
  2. | PMFIdentifier of Ast.AbstractSyntaxTree.identifier
    (*

    A variable name: 'x', 'y', etc

    *)
  3. | PMFApplication of pmf_expression * pmf_expression * pmf_expression list
    (*

    Application: f x

    *)
  4. | PMFLetIn of pmf_case * pmf_expression
    (*

    Let in (without fun and pattern matching): 'let f = 1 and g = 2 in f + g

    *)
  5. | PMFIfThenElse of pmf_expression * pmf_expression * pmf_expression option
    (*

    if condition then true_branch else false branch (else option)

    *)
  6. | PMFTuple of pmf_expression * pmf_expression * pmf_expression list
    (*

    Tuple: '(E1, E2, ..., En)'

    *)
  7. | PMFListConstructor of pmf_expression * pmf_expression
    (*

    List construction: 1 :: 2 ::

    *)
  8. | PMFEmptyList
    (*

    Empty list: ''

    *)
  9. | PMFTyped of pmf_expression * Ast.AbstractSyntaxTree.type_definition
    (*

    Assigning a type to an expression: (expr : int)

    *)
type pmf_decl_case = Ast.AbstractSyntaxTree.identifier * Ast.AbstractSyntaxTree.identifier list * pmf_expression

Binding main identifier * arguments identifiers * expression

type pmf_decl =
  1. | PMFDOrdinary of pmf_decl_case
    (*

    Top-level let-binding: 'let f x = x

    *)
  2. | PMFDRecursive of pmf_decl_case * pmf_decl_case list
    (*

    Top-level recursive let-binding: 'let rec f x = g x and g x = f x'

    *)
type pmf_program = pmf_decl list

The entire lambda lifted code of the program
