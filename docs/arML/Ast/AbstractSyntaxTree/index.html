type case = pattern * expression

Binding pattern with expression

and expression =
  1. | EConstant of constant
    (*

    Any constant: '1', 'true', etc

    *)
  2. | EIdentifier of identifier
    (*

    A variable name: 'x', 'y', etc

    *)
  3. | EFun of pattern * pattern list * expression
    (*

    Anonymous function: 'fun x -> x'

    *)
  4. | EFunction of case * case list
    (*

    Anonymous function with one argument and pattern matching: 'function | hd :: tl -> (hd, tl) | _ -> (0, 0)'

    *)
  5. | EApplication of expression * expression * expression list
    (*

    Application: f x

    *)
  6. | EIfThenElse of expression * expression * expression option
    (*

    if condition then true_branch else false branch (else option)

    *)
  7. | ETuple of expression * expression * expression list
    (*

    Tuple: '(E1, E2, ..., En)'

    *)
  8. | EListConstructor of expression * expression
    (*

    List construction: 1 :: 2 ::

    *)
  9. | EEmptyList
    (*

    Empty list: ''

    *)
  10. | EMatchWith of expression * case * case list
    (*

    Pattern matching: match x with | y -> y

    *)
  11. | ELetIn of case * case list * expression
    (*

    Let in: 'let f x = x in f 5

    *)
  12. | ERecLetIn of case * case list * expression
    (*

    Recursive let in: 'let rec f x = if x = 1 then 1 else x + f (x-1)'

    *)
  13. | ETyped of expression * type_defenition
    (*

    Assigning a type to an expression: (expr : int)

    *)
