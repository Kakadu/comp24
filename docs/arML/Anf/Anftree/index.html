type immut_expr =
  1. | IConstant of Ast.AbstractSyntaxTree.constant
  2. | IEmptyList
  3. | IIdentifier of Ast.AbstractSyntaxTree.identifier
  4. | ITuple of immut_expr * immut_expr * immut_expr list
type complex_expr =
  1. | CAtom of immut_expr
  2. | CApplication of immut_expr * immut_expr * immut_expr list
  3. | CIfThenElse of immut_expr * a_expr * a_expr
  4. | CListConstructor of immut_expr * immut_expr
  5. | CTyped of complex_expr * Ast.AbstractSyntaxTree.type_definition
and a_expr =
  1. | ALetIn of Ast.AbstractSyntaxTree.identifier * complex_expr * a_expr
  2. | AComplex of complex_expr
type anf_decl_case = Ast.AbstractSyntaxTree.identifier * Ast.AbstractSyntaxTree.identifier list * a_expr
type anf_decl =
  1. | ADOrdinary of anf_decl_case
  2. | ADRecursive of anf_decl_case * anf_decl_case list
type anf_program = anf_decl list
