type typ =
  1. | TGround of ground_type
    (*

    Basic types: int, bool, unit, etc.

    *)
  2. | TVar of type_var
    (*

    Polymorphic type variable: 'a, 'b, etc.

    *)
  3. | TArr of typ * typ
    (*

    Function type: 'a -> 'b

    *)
  4. | TTuple of typ list
    (*

    Tuple type: 'a * int * char

    *)
  5. | TList of typ
    (*

    List type: 'a list

    *)
