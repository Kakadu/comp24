type error =
  1. | Occurs_check
    (*

    Trying to unify two types when one of the types contains the other type as a subtype or variable.

    *)
  2. | Unbound_variable of string
    (*

    An undeclared variable is used.

    *)
  3. | Unification_failed of TypeTree.typ * TypeTree.typ
    (*

    Castable types are not compatible.

    *)
  4. | InvalidRecursionLeftHand
    (*

    The left-hand side of a recursive let binding is not a simple variable. e.g., let (x, y) = ... in ...

    *)
  5. | Several_bounds of string
    (*

    A type variable is assigned more than one constraint.

    *)
