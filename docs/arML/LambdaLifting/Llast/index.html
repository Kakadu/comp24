type ll_case = Ast.AbstractSyntaxTree.pattern * ll_expr

Binding pattern -> expression

and ll_expr =
  1. | LEConstant of Ast.AbstractSyntaxTree.constant
    (*

    Any constant: '1', 'true', etc

    *)
  2. | LEIdentifier of Ast.AbstractSyntaxTree.identifier
    (*

    A variable name: 'x', 'y', etc

    *)
  3. | LEApplication of ll_expr * ll_expr * ll_expr list
    (*

    Application: f x

    *)
  4. | LELetIn of ll_case * ll_case list * ll_expr
    (*

    Let in (without fun): 'let f = 1 and g = 2 in f + g

    *)
  5. | LEIfThenElse of ll_expr * ll_expr * ll_expr option
    (*

    if condition then true_branch else false branch (else option)

    *)
  6. | LETuple of ll_expr * ll_expr * ll_expr list
    (*

    Tuple: '(E1, E2, ..., En)'

    *)
  7. | LEEmptyList
    (*

    Empty list: ''

    *)
  8. | LEListConstructor of ll_expr * ll_expr
    (*

    List construction: 1 :: 2 ::

    *)
  9. | LEMatchWith of ll_expr * ll_case * ll_case list
    (*

    Pattern matching: match x with | y -> y

    *)
  10. | LETyped of ll_expr * Ast.AbstractSyntaxTree.type_definition
    (*

    Assigning a type to an expression: (expr : int)

    *)
type ll_decl_case = Ast.AbstractSyntaxTree.pattern * Ast.AbstractSyntaxTree.pattern list * ll_expr

Binding mainPattern * patterns * expression

and ll_decl =
  1. | LDOrdinary of ll_decl_case * ll_decl_case list
    (*

    Top-level let-binding: 'let f x = x and g x = x'

    *)
  2. | LDRecursive of ll_decl_case * ll_decl_case list
    (*

    Top-level recursive let-binding: 'let rec f x = g x and g x = f x'

    *)
type ll_program = ll_decl list

The entire lambda lifted code of the program
