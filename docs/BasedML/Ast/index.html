type typeName =
  1. | TInt

  3. | TPoly of string
  4. | TTuple of typeName list
  5. | TFunction of typeName * typeName
  6. | TList of typeName
val pp_typeName : Ppx_deriving_runtime.Format.formatter -> typeName -> Ppx_deriving_runtime.unit
val show_typeName : typeName -> Ppx_deriving_runtime.string
type rec_flag =
  1. | Rec
  2. | NotRec
val pp_rec_flag : Ppx_deriving_runtime.Format.formatter -> rec_flag -> Ppx_deriving_runtime.unit
val show_rec_flag : rec_flag -> Ppx_deriving_runtime.string
type constant =
  1. | CInt of int
  2. | CBool of bool
val pp_constant : Ppx_deriving_runtime.Format.formatter -> constant -> Ppx_deriving_runtime.unit
val show_constant : constant -> Ppx_deriving_runtime.string
type pattern_no_constraint =
  1. | PWildCard
  2. | PNil
  3. | PCons of pattern_no_constraint * pattern_no_constraint
  4. | PIdentifier of string
  5. | PTuple of pattern_no_constraint list
  6. | PConstant of constant
and pattern =
  1. | PConstraint of pattern_no_constraint * typeName
  2. | PNConstraint of pattern_no_constraint
val pp_pattern_no_constraint : Ppx_deriving_runtime.Format.formatter -> pattern_no_constraint -> Ppx_deriving_runtime.unit
val show_pattern_no_constraint : pattern_no_constraint -> Ppx_deriving_runtime.string
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expr =
  1. | EConstant of constant
  2. | EIdentifier of string
  3. | ENil

  5. | EApplication of expr * expr
  6. | EIfThenElse of expr * expr * expr
  7. | ELetIn of rec_flag * pattern * expr * expr
  8. | ETuple of expr list
  9. | EMatch of pattern * (pattern * expr) list
  10. | EConstraint of expr * typeName
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string

  2. | DMutualRecDecl of singleLet list

val show_singleLet : singleLet -> Ppx_deriving_runtime.string
val pp_let_declaration : Ppx_deriving_runtime.Format.formatter -> let_declaration -> Ppx_deriving_runtime.unit

val pp_declarations : Ppx_deriving_runtime.Format.formatter -> declarations -> Ppx_deriving_runtime.unit
val show_declarations : declarations -> Ppx_deriving_runtime.string
