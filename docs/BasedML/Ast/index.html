type type_name =
  1. | TUnit
  2. | TInt
  3. | TBool
  4. | TPoly of string
  5. | TTuple of type_name list
  6. | TFunction of type_name * type_name
  7. | TList of type_name
val show_type_name : type_name -> Ppx_deriving_runtime.string
val pp_type_name : Stdlib.Format.formatter -> type_name -> unit
type rec_flag =
  1. | Rec
  2. | NotRec
val pp_rec_flag : Ppx_deriving_runtime.Format.formatter -> rec_flag -> Ppx_deriving_runtime.unit
val show_rec_flag : rec_flag -> Ppx_deriving_runtime.string
type constant =
  1. | CInt of int
  2. | CBool of bool
  3. | CNil
  4. | CUnit
val pp_constant : Ppx_deriving_runtime.Format.formatter -> constant -> Ppx_deriving_runtime.unit
val show_constant : constant -> Ppx_deriving_runtime.string
type pattern =
  1. | PWildCard
  2. | PCons of pattern * pattern
  3. | PIdentifier of string
  4. | PTuple of pattern list
    (*

    List.length >= 2

    *)
  5. | PConstant of constant
  6. | PConstraint of pattern * type_name
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expr =
  1. | EConstant of constant
  2. | EIdentifier of string
  3. | EFunction of pattern * expr
  4. | EApplication of expr * expr
  5. | EIfThenElse of expr * expr * expr
  6. | ELetIn of rec_flag * pattern * expr * expr
  7. | ETuple of expr list
    (*

    List.length >= 2

    *)
  8. | EMatch of expr * (pattern * expr) list
  9. | EConstraint of expr * type_name
val pp_expr : Ppx_deriving_runtime.Format.formatter -> expr -> Ppx_deriving_runtime.unit
val show_expr : expr -> Ppx_deriving_runtime.string
type single_let =
  1. | DLet of pattern * expr
val pp_single_let : Ppx_deriving_runtime.Format.formatter -> single_let -> Ppx_deriving_runtime.unit
val show_single_let : single_let -> Ppx_deriving_runtime.string
type let_declaration =
  1. | DSingleLet of rec_flag * single_let
  2. | DMutualRecDecl of rec_flag * single_let list
    (*

    List.length >= 2

    *)
val pp_let_declaration : Ppx_deriving_runtime.Format.formatter -> let_declaration -> Ppx_deriving_runtime.unit
val show_let_declaration : let_declaration -> Ppx_deriving_runtime.string
type declarations = let_declaration list
val pp_declarations : Ppx_deriving_runtime.Format.formatter -> declarations -> Ppx_deriving_runtime.unit
val show_declarations : declarations -> Ppx_deriving_runtime.string
