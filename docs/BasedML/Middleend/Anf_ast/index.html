type immexpr =
  1. | ImmInt of int
  2. | ImmIdentifier of string
  3. | ImmBool of bool
  4. | ImmUnit
  5. | ImmNil
type cexpr =
  1. | CApplication of immexpr * immexpr
  2. | CIfThenElse of immexpr * aexpr * aexpr
  3. | CTuple of immexpr list
  4. | CMatch of Ast.pattern * (Ast.pattern * aexpr) list
  5. | CImmExpr of immexpr
  6. | CConstraint of immexpr * Ast.type_name
and aexpr =
  1. | ALetIn of Ast.pattern * cexpr * aexpr
  2. | ACExpr of cexpr
type single_anf_binding =
  1. | ALet of Ast.pattern * Ast.pattern list * aexpr
type anf_decl =
  1. | ADSingleLet of Ast.rec_flag * single_anf_binding
  2. | ADMutualRecDecl of Ast.rec_flag * single_anf_binding list
    (*

    List.length >= 2

    *)
type anf_prog = anf_decl list
