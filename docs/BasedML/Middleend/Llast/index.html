type llexpr =
  1. | LLConstant of Ast.constant
  2. | LLIdentifier of string
  3. | LLIfThenElse of llexpr * llexpr * llexpr
  4. | LLApplication of llexpr * llexpr
  5. | LLLetIn of Ast.rec_flag * Ast.pattern * llexpr * llexpr
  6. | LLConstraint of llexpr * Ast.type_name
  7. | LLTuple of llexpr list
  8. | LLMatch of Ast.pattern * (Ast.pattern * llexpr) list
val pp_llexpr : Ppx_deriving_runtime.Format.formatter -> llexpr -> Ppx_deriving_runtime.unit
val show_llexpr : llexpr -> Ppx_deriving_runtime.string
type llbinding =
  1. | LLLet of Ast.pattern * Ast.pattern list * llexpr
val pp_llbinding : Ppx_deriving_runtime.Format.formatter -> llbinding -> Ppx_deriving_runtime.unit
val show_llbinding : llbinding -> Ppx_deriving_runtime.string
type lllet_declaration =
  1. | LLDSingleLet of Ast.rec_flag * llbinding
  2. | LLDMutualRecDecl of Ast.rec_flag * llbinding list
    (*

    List.length >= 2

    *)
val pp_lllet_declaration : Ppx_deriving_runtime.Format.formatter -> lllet_declaration -> Ppx_deriving_runtime.unit
val show_lllet_declaration : lllet_declaration -> Ppx_deriving_runtime.string
type llprogram = lllet_declaration list
val pp_llprogram : Ppx_deriving_runtime.Format.formatter -> llprogram -> Ppx_deriving_runtime.unit
val show_llprogram : llprogram -> Ppx_deriving_runtime.string
