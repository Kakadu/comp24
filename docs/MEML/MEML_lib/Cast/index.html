type cexpression =
  1. | CConst of Ast.const
  2. | CVar of Ast.name
  3. | CIfElse of cexpression * cexpression * cexpression
  4. | CEbinOp of Ast.binary_op * cexpression * cexpression
  5. | CApp of cexpression * cexpression
  6. | CLetIn of Ast.rec_flag * Ast.name * Ast.pattern list * cexpression * cexpression
  7. | CPatLetIn of Ast.pattern * cexpression * cexpression
  8. | CTuple of cexpression list
  9. | CMatch of cexpression * (Ast.pattern * cexpression) list
  10. | CList of cexpression * cexpression
val pp_cexpression : Ppx_deriving_runtime.Format.formatter -> cexpression -> Ppx_deriving_runtime.unit
val show_cexpression : cexpression -> Ppx_deriving_runtime.string
type cbindings =
  1. | CLets of Ast.rec_flag * clets list
  2. | CExpression of cexpression
    (*

    simple expressions

    *)
and clets =
  1. | CLet of Ast.name * Ast.pattern list * cexpression
    (*

    let id = expr

    *)
  2. | CLetPat of Ast.pattern * cexpression
    (*

    let id = expr

    *)
val pp_cbindings : Ppx_deriving_runtime.Format.formatter -> cbindings -> Ppx_deriving_runtime.unit
val show_cbindings : cbindings -> Ppx_deriving_runtime.string
val pp_clets : Ppx_deriving_runtime.Format.formatter -> clets -> Ppx_deriving_runtime.unit
val show_clets : clets -> Ppx_deriving_runtime.string
type cstatements = cbindings list
val pp_cstatements : Ppx_deriving_runtime.Format.formatter -> cstatements -> Ppx_deriving_runtime.unit
val show_cstatements : cstatements -> Ppx_deriving_runtime.string
