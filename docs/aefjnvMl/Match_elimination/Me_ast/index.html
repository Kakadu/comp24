type 'a id_t =
  1. | Id_name of 'a
  2. | Id_unit
type m_expr =
  1. | MExp_constant of Common.Ast.const
  2. | MExp_ident of Common.Ast.ident
  3. | MExp_tuple of m_expr list
  4. | MExp_apply of m_expr * m_expr
  5. | MExp_list of m_expr * m_expr
  6. | MExp_ifthenelse of m_expr * m_expr * m_expr
  7. | MExp_function of string id_t * m_expr
  8. | MExp_let of m_decl * m_expr
and m_decl =
  1. | MDecl of Common.Ast.rec_flag * value_binding list
and value_binding = {
  1. m_vb_pat : string id_t;
  2. m_vb_expr : m_expr;
}
type m_program = m_decl list
