type parse_error =
  1. | Syntax_error of string
type infer_error =
  1. | Occurs_check of int * Typedtree.ty
    (*

    This error can lead to non-terminate of the type inference Example: fun x -> x x

    *)
  2. | Unbound_variable of string
    (*

    Variable is not defined in local context Example: let increase = x + 1

    *)
  3. | Unification_failed of Typedtree.ty * Typedtree.ty
    (*

    Unification error between two types

    *)
  4. | Several_bounds of string
    (*

    Example: let (x, x) = 1, 3

    *)
  5. | Not_specify_rec
  6. | No_variable_rec
    (*

    Only variables are allowed as left-side of 'let rec' Example: let rec (a, b) = 5

    *)
type error =
  1. | Parser of parse_error
  2. | Infer of infer_error
val occurs_check : (Typedtree.binder * Typedtree.ty) -> error
val unbound_variable : string -> error
val unification_failed : (Typedtree.ty * Typedtree.ty) -> error
val several_bounds : string -> error
val not_specify_rec : error
val no_variable_rec : error
