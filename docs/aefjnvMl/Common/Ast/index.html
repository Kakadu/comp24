type ident = string
val pp_ident : Ppx_deriving_runtime.Format.formatter -> ident -> Ppx_deriving_runtime.unit
val show_ident : ident -> Ppx_deriving_runtime.string
type rec_flag =
  1. | Nonrecursive
  2. | Recursive
val equal_rec_flag : rec_flag -> rec_flag -> Ppx_deriving_runtime.bool
val pp_rec_flag : Ppx_deriving_runtime.Format.formatter -> rec_flag -> Ppx_deriving_runtime.unit
val show_rec_flag : rec_flag -> Ppx_deriving_runtime.string
type const =
  1. | Const_int of int
    (*

    Integers constants such as 52

    *)
  2. | Const_bool of bool
    (*

    Boolean constant: true, false

    *)
  3. | Const_nil
    (*

    Represents empty list []

    *)
  4. | Const_unit
val pp_const : Ppx_deriving_runtime.Format.formatter -> const -> Ppx_deriving_runtime.unit
val show_const : const -> Ppx_deriving_runtime.string
type core_type =
  1. | Ptyp_int
    (*

    Integer type int

    *)
  2. | Ptyp_bool
    (*

    Boolean type bool

    *)
  3. | Ptyp_unit
    (*

    Unit type unit

    *)
  4. | Ptyp_var of ident
    (*

    A type variable such as 'a

    *)
  5. | Ptyp_list of core_type
    (*

    List type such as int list

    *)
  6. | Ptyp_tuple of core_type list
    (*

    Ptyp_tuple([T1 ; ... ; Tn]) represents a product type T1 * ... * Tn. Invariant: n >= 2.

    *)
  7. | Ptyp_arrow of core_type * core_type
    (*

    Ptyp_arrow(lbl, T1, T2) represents: T1 -> T2

    *)
val pp_core_type : Ppx_deriving_runtime.Format.formatter -> core_type -> Ppx_deriving_runtime.unit
val show_core_type : core_type -> Ppx_deriving_runtime.string
type pattern =
  1. | Pat_any
    (*

    The pattern _

    *)
  2. | Pat_var of ident
    (*

    A variable pattern such as x

    *)
  3. | Pat_const of const
    (*

    Patterns such as 1, true

    *)
  4. | Pat_cons of pattern * pattern
    (*

    The pattern such as P1::P2

    *)
  5. | Pat_tuple of pattern list
    (*

    Patterns (P1, ..., P2) Invariant : n >= 2

    *)
  6. | Pat_constraint of pattern * core_type
val pp_pattern : Ppx_deriving_runtime.Format.formatter -> pattern -> Ppx_deriving_runtime.unit
val show_pattern : pattern -> Ppx_deriving_runtime.string
type expression =
  1. | Exp_type of expression * core_type
  2. | Exp_constant of const
    (*

    Expressions constant such as 1, true

    *)
  3. | Exp_ident of ident
    (*

    Identifiers such as x

    *)
  4. | Exp_tuple of expression list
    (*

    Expressions (E1, ..., En) Invariant: n >= 2

    *)
  5. | Exp_function of pattern * expression
    (*

    fun P1 -> E

    *)
  6. | Exp_let of decl * expression
    (*

    Exp_let({d_rec=flag; P; E}, E') represents:

    • let P = E in E' when flag is Nonrecursive
    • let rec P = E in E' when d_rec is Recursive
    *)
  7. | Exp_match of expression * (pattern * expression) list
    (*

    match E0 with P1 -> E1 | .. | Pn -> En

    *)
  8. | Exp_ifthenelse of expression * expression * expression
    (*

    if E1 then E2 else E3

    *)
  9. | Exp_apply of expression * expression
    (*

    E0 E1 Invariant: n > 1

    *)
  10. | Exp_list of expression * expression
    (*

    The expression such as E1::E2 This also represents lists E1; ... En via E

    *)
and decl =
  1. | Decl of rec_flag * value_binding list

Represents:

  • let P = E when d_rec is Nonrecursive
  • let rec P = E when d_rec is Recursive Invariant: n >= 1
and value_binding = {
  1. vb_pat : pattern;
  2. vb_expr : expression;
}
val pp_expression : Ppx_deriving_runtime.Format.formatter -> expression -> Ppx_deriving_runtime.unit
val show_expression : expression -> Ppx_deriving_runtime.string
val pp_decl : Ppx_deriving_runtime.Format.formatter -> decl -> Ppx_deriving_runtime.unit
val show_decl : decl -> Ppx_deriving_runtime.string
val pp_value_binding : Ppx_deriving_runtime.Format.formatter -> value_binding -> Ppx_deriving_runtime.unit
val show_value_binding : value_binding -> Ppx_deriving_runtime.string
type structure_item =
  1. | Str_eval of expression
    (*

    E

    *)
  2. | Str_value of decl
    (*

    Str_value({is_rec; P; E}) represents:

    • let P = E when Nonrecursive - let rec P = E when d_rec is Recursive
    *)
val pp_structure_item : Ppx_deriving_runtime.Format.formatter -> structure_item -> Ppx_deriving_runtime.unit
val show_structure_item : structure_item -> Ppx_deriving_runtime.string
type program = structure_item list

Represents whole program with all statements

val pp_program : Ppx_deriving_runtime.Format.formatter -> program -> Ppx_deriving_runtime.unit
val show_program : program -> Ppx_deriving_runtime.string
