type 'a scoped =
  1. | Local_id of 'a
  2. | Global_func of 'a
  3. | Global_var of 'a
type immexpr =
  1. | Imm_int of int
  2. | Imm_bool of bool
  3. | Imm_nil
  4. | Imm_unit
  5. | Imm_id of string scoped Match_elimination.Me_ast.id_t
type cexpr =
  1. | C_immexpr of immexpr
  2. | C_apply of string scoped * immexpr * immexpr list
  3. | C_tuple of immexpr list
  4. | C_rlist of immexpr * immexpr list
    (*

    reversed list <- first elem -- nil

    *)
  5. | C_ifthenelse of immexpr * aexpr * aexpr
and aexpr =
  1. | A_let of string Match_elimination.Me_ast.id_t * cexpr * aexpr
  2. | A_cexpr of cexpr
type anf_fun = {
  1. adec_name : string;
  2. adec_args : string Match_elimination.Me_ast.id_t * string Match_elimination.Me_ast.id_t list;
  3. adec_body : aexpr;
}
type anf_decl =
  1. | A_GlobalV of string Match_elimination.Me_ast.id_t * aexpr
  2. | A_NonrecDecl of anf_fun
  3. | A_RecDecl of anf_fun * anf_fun list
type anf_prog = anf_decl list
