type immexpr =
  1. | ImmId of string
  2. | ImmInt of int
  3. | ImmString of string
  4. | ImmBool of bool
  5. | ImmEmptyList
  6. | ImmUnit
  7. | ImmTuple of immexpr list
type cexpr =
  1. | CImmExpr of immexpr
  2. | CApp of string * immexpr list
  3. | CIf of immexpr * aexpr * aexpr
  4. | CCons of immexpr * immexpr
and aexpr =
  1. | ALet of string * cexpr * aexpr
  2. | ACExpr of cexpr
type tp = bool * (string * string list * aexpr) list
type program = tp list
