type pattern =
  1. | Pat_any
    (*

    The pattern _

    *)
  2. | Pat_var of Smisc.Ident.t
    (*

    A variable pattern such as x

    *)
  3. | Pat_constraint of pattern * ty
    (*

    (P : T)

    *)
  4. | Pat_constant of constant
    (*

    Patterns such as 1, 'a', "hello", 1.5

    *)
  5. | Pat_tuple of pattern Smisc.list2
    (*

    Patterns (P1, ..., Pn). Invariant: n >= 2

    *)
  6. | Pat_construct of Smisc.Ident.t * pattern Base.option
    (*

    Pat_construct(C, args) represents:

    • C when args is None
    • C P when args is Some P
    *)

and expression =
  1. | Exp_ident of Smisc.Ident.t
    (*

    Identifiers such as x, fact

    *)
  2. | Exp_constraint of expression * ty
    (*

    (E : T)

    *)
  3. | Exp_constant of constant
    (*

    Expression constant such as 1, 'a', "hello", 1.5

    *)
  4. | Exp_let of rec_flag * value_binding Smisc.list1 * expression
    (*

    Exp_let(flag, [(P1,E1) ; ... ; (Pn,En)], E) represents:

    • let P1 = E1 and ... and Pn = EN in E when flag is Nonrecursive
    • let rec P1 = E1 and ... and Pn = EN in E when flag is Recursive. Invariant: n >= 1
    *)
  5. | Exp_fun of pattern Smisc.list1 * expression
    (*

    Exp_fun ([P1; ...; Pn], E) represents fun P1 ... Pn -> E. Invariant: n >= 1

    *)
  6. | Exp_apply of expression * expression
    (*

    Exp_apply(E0, E1) represents E0 E1

    *)
  7. | Exp_match of expression * case Smisc.list1
    (*

    match E0 with P1 -> E1 | ... | Pn -> En. Invariant: n >= 1

    *)
  8. | Exp_tuple of expression Smisc.list2
    (*

    Expressions (E1, ..., En). Invariant: n >= 2

    *)
  9. | Exp_construct of Smisc.Ident.t * expression Base.option
    (*

    Exp_construct(C, exp) represents:

    • C when exp is None,
    • C E when exp is Some E,
    • C (E1, ..., En) when exp is Some (Exp_tuple[E1;...;En])
    *)
  10. | Exp_ifthenelse of expression * expression * expression Base.option
    (*

    if E1 then E2 else E3

    *)
