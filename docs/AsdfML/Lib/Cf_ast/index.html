type cf_expr =
  1. | CFConst of Ast.constant
  2. | CFVar of Ast.id
  3. | CFApp of cf_expr * cf_expr
  4. | CFIfElse of cf_expr * cf_expr * cf_expr
  5. | CFLetIn of Ast.id * cf_expr * cf_expr
  6. | CFTuple of cf_expr * cf_expr * cf_expr list
  7. | CFList of cf_expr list
and cf_definition =
  1. | CFLet of Ast.id * Ast.id list * cf_expr
type program = cf_definition list
val cf_const : Ast.constant -> cf_expr
val cf_var : Ast.id -> cf_expr
val cf_app : cf_expr -> cf_expr -> cf_expr
val cf_if_else : cf_expr -> cf_expr -> cf_expr -> cf_expr
val cf_let_in : Ast.id -> cf_expr -> cf_expr -> cf_expr
val cf_tuple : cf_expr -> cf_expr -> cf_expr list -> cf_expr
val cf_list : cf_expr list -> cf_expr
val cf_def : Ast.id -> Ast.id list -> cf_expr -> cf_definition
val pp_expr : Base.Format.formatter -> cf_expr -> unit
val pp_definition : Base.Format.formatter -> cf_definition -> unit
val pp_program : Base.Format.formatter -> cf_definition Base.List.t -> unit
