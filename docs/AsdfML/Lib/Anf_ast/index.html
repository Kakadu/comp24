type id = string
type imm_expr =
  1. | ImmInt of int
  2. | ImmBool of bool
  3. | ImmUnit
  4. | ImmNil
  5. | ImmId of id
  6. | ImmTuple of imm_expr * imm_expr * imm_expr list
  7. | ImmList of imm_expr list
val show_imm_expr : imm_expr -> Ppx_deriving_runtime.string
type cexpr =
  1. | CApp of imm_expr * imm_expr list
  2. | CIfElse of imm_expr * aexpr * aexpr
  3. | CImmExpr of imm_expr
and aexpr =
  1. | ALet of id * cexpr * aexpr
  2. | ACExpr of cexpr
val show_cexpr : cexpr -> Ppx_deriving_runtime.string
val show_aexpr : aexpr -> Ppx_deriving_runtime.string
type fn =
  1. | Fn of Ast.rec_flag * id * id list * aexpr
val show_fn : fn -> Ppx_deriving_runtime.string
type program = fn list
val show_program : program -> Ppx_deriving_runtime.string
val count_bindings : fn -> int
val pp_imm_expr : Base.Format.formatter -> imm_expr -> unit
val pp_cexpr : Base.Format.formatter -> cexpr -> unit
val pp_aexpr : Base.Format.formatter -> aexpr -> unit
val pp_fn : Base.Format.formatter -> fn -> unit
val pp_program : Base.Format.formatter -> fn list -> unit
