open Ast

let print_int = "print_int"
let getfield = "getfield"
let stdlib_funs = [ print_int; getfield ]

let is_stdlib_fun f =
  match List.find_opt (( = ) f) stdlib_funs with
  | Some _ -> true
  | None -> false
;;
